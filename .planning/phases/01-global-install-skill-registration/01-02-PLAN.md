    ```typescript
    #!/usr/bin/env node

    import { readFileSync } from 'node:fs';

    const args = process.argv.slice(2);
    const command = args[0];

    switch (command) {
      case 'install-skills': {
        const { registerSkills } = await import('../src/register-skills.js');
        await registerSkills();
        console.log('id3-cli: Skills registered at ~/.claude/skills/');
        break;
      }

      case 'uninstall-skills': {
        const { unregisterSkills } = await import('../src/register-skills.js');
        await unregisterSkills();
        console.log('id3-cli: Skills unregistered from ~/.claude/skills/');
        break;
      }

      case '--help':
      case '-h': {
        printHelp();
        break;
      }

      case '--version':
      case '-v': {
        console.log(getVersion());
        break;
      }

      case 'init':
      default: {
        // Existing init flow (backward compat for npx id3-cli .)
        const { initProject, printBanner, printSuccess } = await import('../src/init.js');
        const { createInterface } = await import('node:readline');

        const flags = args.filter((a) => a.startsWith('--'));
        const positional = args.filter((a) => !a.startsWith('--'));

        const hasInit = positional[0] === 'init';
        const targetDir = hasInit ? (positional[1] ?? '.') : (positional[0] ?? '.');
        const noSymlink = flags.includes('--no-symlink');

        printBanner();

        const result = await initProject(targetDir, { noSymlink });

        if (result.alreadyInstalled) {
          const answer = await prompt(
            'IDDD appears to be already installed. Overwrite? (y/N) ',
            createInterface
          );
          if (answer.toLowerCase() !== 'y') {
            console.log('Aborted.');
            process.exit(0);
          }
          const retryResult = await initProject(targetDir, {
            overwrite: true,
            noSymlink,
          });
          printSuccess(retryResult);
        } else {
          printSuccess(result);
        }
        break;
      }
    }

    function printHelp(): void {
      console.log(`Usage: id3-cli [command] [options]

    Commands:
      init [dir]           Initialize IDDD in a project (default command)
      install-skills       Register id3-start and id3-clear skills at ~/.claude/skills/
      uninstall-skills     Remove registered skills from ~/.claude/skills/
                           Run this before \`npm uninstall -g id3-cli\`

    Options:
      --no-symlink         Copy skill files instead of creating symlinks
      -v, --version        Show version number
      -h, --help           Show this help message

    Examples:
      npx id3-cli .                  Initialize IDDD in current directory
      id3-cli init ./my-project      Initialize IDDD in a specific directory
      id3-cli install-skills         Manually register global skills
      id3-cli uninstall-skills       Clean up skills before uninstalling`);
    }

    function getVersion(): string {
      try {
        const pkgPath = new URL('../../package.json', import.meta.url);
        const pkg = JSON.parse(readFileSync(pkgPath, 'utf-8'));
        return pkg.version ?? '0.0.0';
      } catch {
        return '0.0.0';
      }
    }

    function prompt(
      question: string,
      createInterface: typeof import('node:readline').createInterface
    ): Promise<string> {
      const rl = createInterface({ input: process.stdin, output: process.stdout });
      return new Promise((resolve) => {
        rl.question(question, (answer) => {
          rl.close();
          resolve(answer);
        });
      });
    }
    ```

    Key design decisions:
    - Use dynamic `await import()` for register-skills.js and init.js to avoid loading unnecessary modules for each subcommand
    - The default case handles both `id3-cli .` (no subcommand, backward compat) and `id3-cli init .` (explicit init)
    - `--help` and `--version` only match when they are the first argument (i.e., `command` position). `--no-symlink` is a flag consumed by the init flow, not a top-level command.
    - `getVersion()` function uses the same `import.meta.url` pattern as the existing code in `src/init.ts`
    - `prompt` function takes `createInterface` as a parameter so it is only imported in the init flow

    ```typescript
    import { registerSkills } from '../src/register-skills.js';

    try {
      await registerSkills();
      console.log('id3-cli: Skills registered at ~/.claude/skills/');
      console.log('  /id3-start and /id3-clear are now available in Claude Code.');
    ```

    ```typescript
    import { describe, it, expect, beforeAll } from 'vitest';
    import { execSync } from 'node:child_process';
    import { join } from 'node:path';

    const CLI_PATH = join(import.meta.dirname, '..', 'dist', 'bin', 'cli.js');

    function runCli(args: string): string {
      return execSync(`node ${CLI_PATH} ${args}`, {
        encoding: 'utf-8',
        timeout: 5000,
      }).trim();
    }

    describe('CLI subcommand routing', () => {
      beforeAll(() => {
        // Ensure dist/ is up to date
        execSync('npx tsc', {
          cwd: join(import.meta.dirname, '..'),
          encoding: 'utf-8',
        });
      });

      it('--version prints a version string', () => {
        const output = runCli('--version');
        expect(output).toMatch(/^\d+\.\d+\.\d+/);
      });

      it('-v prints the same version as --version', () => {
        const v1 = runCli('--version');
        const v2 = runCli('-v');
        expect(v1).toBe(v2);
      });

      it('--help output contains install-skills', () => {
        const output = runCli('--help');
        expect(output).toContain('install-skills');
      });

      it('--help output contains uninstall-skills', () => {
        const output = runCli('--help');
        expect(output).toContain('uninstall-skills');
      });

      it('--help output contains init', () => {
        const output = runCli('--help');
        expect(output).toContain('init');
      });

      it('-h output matches --help output', () => {
        const help1 = runCli('--help');
        const help2 = runCli('-h');
        expect(help1).toBe(help2);
      });
    });
    ```
